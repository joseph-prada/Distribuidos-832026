server/config/db.js — Conexión Oracle Oracle

const oracledb = require('oracledb');
const path = require('path');

// Inicializar cliente Oracle Thick (necesario para Wallet)
oracledb.initOracleClient({ libDir: process.env.ORACLE_LIB_DIR });

const dbConfig = {
  user:             process.env.DB_USER,
  password:         process.env.DB_PASSWORD,
  connectString:    process.env.DB_CONNECT_STRING, // ej: mydb_high
  walletLocation:   path.join(__dirname, '../../wallet'),
  walletPassword:   process.env.WALLET_PASSWORD,
};

let pool;

async function initPool() {
  pool = await oracledb.createPool({
    ...dbConfig,
    poolMin:  2,
    poolMax:  10,
    poolIncrement: 1,
  });
  console.log('✓ Oracle connection pool ready');
}

async function getConnection() {
  return await pool.getConnection();
}

module.exports = { initPool, getConnection };

server/services/emailService.js Service

const { getConnection } = require('../config/db');
const transporter    = require('../config/mailer');

async function sendEmail({ from, to, subject, body, userId }) {
  // 1. Enviar el correo real vía SMTP
  const info = await transporter.sendMail({
    from, to, subject,
    html: body,
  });

  // 2. Guardar en Oracle
  const conn = await getConnection();
  try {
    const result = await conn.execute(
      `INSERT INTO EMAILS
         (ID, FROM_ADDRESS, TO_ADDRESS, SUBJECT, BODY,
          STATUS, MESSAGE_ID, SENT_AT, USER_ID)
       VALUES
         (EMAILS_SEQ.NEXTVAL, :from, :to, :subject, :body,
          'SENT', :messageId, SYSDATE, :userId)`,
      { from, to, subject, body,
        messageId: info.messageId, userId },
      { autoCommit: true }
    );
    return result.lastRowid;
  } finally {
    await conn.close();
  }
}

async function getEmails({ userId, page = 1, limit = 20, search = '' }) {
  const offset = (page - 1) * limit;
  const conn = await getConnection();
  try {
    const { rows } = await conn.execute(
      `SELECT ID, FROM_ADDRESS, TO_ADDRESS, SUBJECT,
              SUBSTR(BODY, 1, 200) AS PREVIEW,
              STATUS, SENT_AT
       FROM EMAILS
       WHERE USER_ID = :userId
         AND (SUBJECT LIKE :search OR TO_ADDRESS LIKE :search)
       ORDER BY SENT_AT DESC
       OFFSET :offset ROWS FETCH NEXT :limit ROWS ONLY`,
      { userId, search: `%${search}%`, offset, limit },
      { outFormat: oracledb.OUT_FORMAT_OBJECT }
    );
    return rows;
  } finally {
    await conn.close();
  }
}

module.exports = { sendEmail, getEmails };

server/routes/emails.js Router

const router         = require('express').Router();
const { sendEmail, getEmails } = require('../services/emailService');
const authMiddleware = require('../middleware/auth');

// Todas las rutas requieren autenticación
router.use(authMiddleware);

// GET /api/emails — listar con paginación y búsqueda
router.get('/', async (req, res) => {
  const { page, limit, search } = req.query;
  const emails = await getEmails({
    userId: req.user.id,
    page:   parseInt(page) || 1,
    limit:  parseInt(limit) || 20,
    search: search || ''
  });
  res.json({ success: true, data: emails });
});

// POST /api/emails/send — enviar correo
router.post('/send', async (req, res) => {
  const { to, subject, body } = req.body;
  if (!to || !subject || !body)
    return res.status(400).json({ error: 'Faltan campos requeridos' });

  const rowId = await sendEmail({
    from:   process.env.SMTP_FROM,
    to, subject, body,
    userId: req.user.id
  });
  res.json({ success: true, id: rowId });
});

// GET /api/emails/:id — ver correo completo
router.get('/:id', async (req, res) => { /* ... */ });

module.exports = router;

04 — Esquema de Base de Datos Oracle

DDL — Creación de Tablas SQL

-- =============================================
-- TABLA: USERS
-- =============================================
CREATE TABLE USERS (
  ID            NUMBER         GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  EMAIL         VARCHAR2(255)  NOT NULL UNIQUE,
  PASSWORD_HASH VARCHAR2(255)  NOT NULL,
  DISPLAY_NAME  VARCHAR2(100),
  CREATED_AT    TIMESTAMP      DEFAULT SYSTIMESTAMP,
  IS_ACTIVE     NUMBER(1)      DEFAULT 1
);

-- =============================================
-- SECUENCIA + TABLA: EMAILS
-- =============================================
CREATE SEQUENCE EMAILS_SEQ START WITH 1 INCREMENT BY 1;

CREATE TABLE EMAILS (
  ID            NUMBER         PRIMARY KEY,
  USER_ID       NUMBER         NOT NULL REFERENCES USERS(ID),
  FROM_ADDRESS  VARCHAR2(255)  NOT NULL,
  TO_ADDRESS    VARCHAR2(2000) NOT NULL,    -- múltiples destinatarios
  CC_ADDRESS    VARCHAR2(2000),
  SUBJECT       VARCHAR2(1000) NOT NULL,
  BODY          CLOB,                          -- HTML permitido
  STATUS        VARCHAR2(20)   DEFAULT 'SENT',-- SENT | DRAFT | FAILED
  MESSAGE_ID    VARCHAR2(500),                 -- ID del servidor SMTP
  SENT_AT       TIMESTAMP      DEFAULT SYSTIMESTAMP,
  IS_READ       NUMBER(1)      DEFAULT 0,
  IS_DELETED    NUMBER(1)      DEFAULT 0
);

-- ÍNDICES para rendimiento
CREATE INDEX IDX_EMAILS_USER     ON EMAILS(USER_ID);
CREATE INDEX IDX_EMAILS_SENT_AT  ON EMAILS(SENT_AT DESC);
CREATE INDEX IDX_EMAILS_STATUS   ON EMAILS(STATUS);

-- =============================================
-- TABLA: ATTACHMENTS
-- =============================================
CREATE TABLE ATTACHMENTS (
  ID          NUMBER          GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  EMAIL_ID    NUMBER          NOT NULL REFERENCES EMAILS(ID) ON DELETE CASCADE,
  FILENAME    VARCHAR2(500)   NOT NULL,
  MIME_TYPE   VARCHAR2(100),
  FILE_SIZE   NUMBER,
  STORAGE_URL VARCHAR2(2000)  -- URL de OCI Object Storage
);

-- GRANT de acceso (si usas esquema aparte)
GRANT SELECT, INSERT, UPDATE, DELETE ON EMAILS TO APP_USER;
GRANT SELECT, INSERT ON USERS TO APP_USER;

06 — Seguridad & Pasos de Deploy

🔐 Seguridad Implementada

  • Conexión Oracle con mTLS vía Wallet (cifrado en tránsito)
  • Contraseñas hasheadas con bcrypt (salt=12)
  • JWT con expiración + refresh token opcional
  • Helmet.js: CSP, HSTS, X-Frame-Options
  • Rate limiting: 100 req/15min por IP
  • Validación y sanitización de todos los inputs
  • Variables sensibles solo en .env (nunca en código)
  • SMTP con TLS obligatorio (STARTTLS)

⚠️ Puntos Críticos a No Olvidar

  • Nunca subir .env ni /wallet/ a git (.gitignore)
  • Gmail requiere "App Password" (no contraseña normal)
  • Oracle Wallet debe ser accesible por el proceso Node
  • CORS restringido a tu dominio/localhost específico
  • Usar HTTPS en producción (Let's Encrypt / Nginx)
  • Body del correo sanitizado contra XSS (DOMPurify)
  • Logs de errores sin datos sensibles
